  • C = number of channels
  • S = number of samples

 213def check_stopping_criterion(
 214    current_time,
 215    nchannels,
 216    current_performance_delta,
 217    max_time,
 218    min_channels,
 219    max_channels,
 220    performance_delta,
 221    print_output=True,
 222):
 223    """Function to check if a stopping criterion has been met.
 224
 225    Parameters
 226    ----------
 227    current_time : float
 228        The time elapsed since the start of the channel selection method.
 229    nchannels : int
 230        The number of channels in the current iteration of the new best channel
 231        subset (`len(new_channel_subset)`).
 232    current_performance_delta : float
 233        The performance delta between the current iteration and the previous.
 234    max_time : int
 235        The maxiumum amount of time, in seconds, that the function will
 236        search for the optimal solution.
 237    min_channels : int
 238        The minimum number of channels.
 239    max_channels : int
 240        The maximum number of channels.
 241    performance_delta : float
 242        The performance delta under which the algorithm is considered to
 243        be close enough to optimal.
 244    print_output : str, *optional*
 245        Flag on whether or not to print output.
 246        - Default is `True`.
 247
 248    Returns
 249    -------
 250    *bool*
 251        Has stopping criterion been met (`True`) or not (`False`).
 252
 253    """
 254    if current_time > max_time:
 255        if print_output == "verbose" or print_output == "final":
 256            print("Stopping based on time")
 257        return True
 258
 259    elif nchannels <= min_channels:
 260        if print_output == "verbose" or print_output == "final":
 261            print("Stopping because minimum number of channels reached")
 262        return True
 263
 264    elif nchannels >= max_channels:
 265        if print_output == "verbose" or print_output == "final":
 266            print("Stopping because maximum number of channels reached")
 267        return True
 268
 269    elif current_performance_delta < performance_delta:
 270        if print_output == "verbose" or print_output == "final":
 271            print("Stopping because performance improvements are declining")
 272        return True
 273    else:
 274        return False

 277def sfs(
 278    kernel_func,
 279    X,
 280    y,
 281    channel_labels,
 282    metric,
 283    initial_channels,
 284    max_time,
 285    min_channels,
 286    max_channels,
 287    performance_delta,
 288    n_jobs,
 289    print_output,
 290    record_performance,
 291):
 292    """
 293    The Sequential Forward Selection (SFS) method for channel selection.
 294
 295    Parameters
 296    ----------
 297    kernel_func : function
 298        The classification kernel function which does feature extraction
 299        and classification.
 300        Different functions  are used for MI, P300, SSVEP, etc.
 301    X : numpy.ndarray
 302        Training data for the classifier as windows of EEG data.
 303        3D array containing data with `float` type.
 304
 305        shape = (`W_windows`,`C_channels`,`S_samples`)
 306    y : numpy.ndarray
 307        Training labels for the classifier.
 308        1D array.
 309
 310        shape = (`nwindows`)
 311    channel_labels: list of `str`
 312        The set of channel labels corresponding to `C_channels`.
 313        A list of strings with length = `C_channels`.
 314    metric : str
 315        The metric used to measure the "goodness" of the trained classifier.
 316    initial_channels : list of `str`
 317        Initial guess of channels.
 318    max_time : int
 319        The maxiumum amount of time, in seconds, that the function will
 320        search for the optimal solution.
 321    min_channels : int
 322        The minimum number of channels.
 323    max_channels : int
 324        The maximum number of channels.
 325    performance_delta : float
 326        The performance delta under which the algorithm is considered to
 327        be close enough to optimal.
 328    n_jobs : int
 329        The number of threads to dedicate to this calculation.
 330    print_output : str
 331        Flag on whether or not to print output. Options are:
 332        - `"verbose"`: Print output at each step.
 333        - `"final"`: Print output at the end.
 334        - `"silent"`: No output.
 335    record_performance : bool
 336        Flag on whether or not to record performance at each step.
 337
 338
 339    Returns
 340    -------
 341    new_channel_subset : list of `str`
 342        The new best channel subset from the list of `channel_labels`.
 343    self.clf : classifier
 344        The trained classification model.
 345    preds : numpy.ndarray
 346        The predictions from the model.
 347        1D array with the same shape as `y`.
 348
 349        shape = (`nwindows`)
 350    accuracy : float
 351        The accuracy of the trained classification model.
 352    precision : float
 353        The precision of the trained classification model.
 354    recall : float
 355        The recall of the trained classification model.
 356    results_df : pandas.DataFrame
 357        The dataframe containing the results of each step of channel selection.
 358    """
 359    results_df = pd.DataFrame(
 360        columns=[
 361            "Step",
 362            "Time",
 363            "N Channels",
 364            "Channel Subset",
 365            "Unique Combinations Tested in Step",
 366            "Accuracy",
 367            "Precision",
 368            "Recall",
 369        ]
 370    )
 371    step = 1
 372
 373    start_time = time.time()
 374
 375    nwindows, nchannels, nsamples = X.shape
 376    sfs_subset = []
 377
 378    for i, c in enumerate(channel_labels):
 379        if c in initial_channels:
 380            sfs_subset.append(i)
 381
 382    previous_performance = 0
 383
 384    stop_criterion = False
 385
 386    # Get the performance of the initial subset, if possible
 387    try:
 388        (
 389            initial_model,
 390            initial_preds,
 391            initial_accuracy,
 392            initial_precision,
 393            initial_recall,
 394        ) = kernel_func(X[:, sfs_subset, :], y)
 395        if metric == "accuracy":
 396            initial_performance = initial_accuracy
 397        elif metric == "precision":
 398            initial_performance = initial_precision
 399        elif metric == "recall":
 400            initial_performance = initial_recall
 401
 402        # Best
 403        best_channel_subset = initial_channels
 404        best_model = initial_model
 405        best_performance = initial_performance
 406        best_preds = initial_preds
 407        best_accuracy = initial_accuracy
 408        best_precision = initial_precision
 409        best_recall = initial_recall
 410
 411    # If not possible then set the initial performance to 0
 412    except ValueError:
 413        best_channel_subset = []
 414        best_model = None
 415        best_performance = 0
 416        best_preds = []
 417        best_accuracy = 0
 418        best_precision = 0
 419        best_recall = 0
 420
 421    preds = []
 422    accuracy = 0
 423    precision = 0
 424    recall = 0
 425
 426    while stop_criterion is False:
 427        sets_to_try = []
 428        X_to_try = []
 429        for c in range(nchannels):
 430            if c not in sfs_subset:
 431                set_to_try = sfs_subset.copy()
 432                set_to_try.append(c)
 433                sets_to_try.append(set_to_try)
 434
 435                # get the new X
 436                new_X = np.zeros((nwindows, len(set_to_try), nsamples))
 437                for i, j in enumerate(set_to_try):
 438                    new_X[:, i, :] = X[:, j, :]
 439
 440                # make a list f all subsets of X to try
 441                X_to_try.append(new_X)
 442
 443        # This handles the multiprocessing to check multiple channel combinations at once if n_jobs > 1
 444        outputs = Parallel(n_jobs=n_jobs)(
 445            delayed(kernel_func)(Xtest, y) for Xtest in X_to_try
 446        )
 447
 448        models = []
 449        predictions = []
 450        accuracies = []
 451        precisions = []
 452        recalls = []
 453
 454        # Extract the outputs
 455        for output in outputs:
 456            models.append(output[0])
 457            predictions.append(output[1])
 458            accuracies.append(output[2])
 459            precisions.append(output[3])
 460            recalls.append(output[4])
 461
 462        # Get the performance metric
 463        if metric == "accuracy":
 464            performances = accuracies
 465        elif metric == "precision":
 466            performances = precisions
 467        elif metric == "recall":
 468            performances = recalls
 469        else:
 470            print("performance metric invalid, defaulting to accuracy")
 471            performances = accuracies
 472
 473        # Get the index of the best X tried in this round
 474        best_set_index = accuracies.index(np.max(performances))
 475
 476        sfs_subset = sets_to_try[best_set_index]
 477        new_channel_subset = [channel_labels[c] for c in sfs_subset]
 478        model = models[best_set_index]
 479        preds = predictions[best_set_index]
 480        accuracy = accuracies[best_set_index]
 481        # best_overall_accuracy = accuracy
 482        precision = precisions[best_set_index]
 483        recall = recalls[best_set_index]
 484        if print_output == "verbose":
 485            print("new subset ", new_channel_subset)
 486            print("accuracy ", accuracy)
 487            print("accuracies ", accuracies)
 488
 489        if metric == "accuracy":
 490            current_performance = accuracy
 491
 492        p_delta = current_performance - previous_performance
 493        previous_performance = current_performance
 494
 495        if current_performance > best_performance:
 496            best_channel_subset = new_channel_subset
 497            best_model = model
 498            best_performance = current_performance
 499            best_preds = preds
 500            best_accuracy = accuracy
 501            best_precision = precision
 502            best_recall = recall
 503        elif current_performance >= best_performance and len(new_channel_subset) < len(
 504            best_channel_subset
 505        ):
 506            best_channel_subset = new_channel_subset
 507            best_model = model
 508            best_performance = current_performance
 509            best_preds = preds
 510            best_accuracy = accuracy
 511            best_precision = precision
 512            best_recall = recall
 513
 514        if record_performance is True:
 515            new_channel_subset.sort()
 516            results_df.loc[step] = [
 517                step,
 518                time.time() - start_time,
 519                len(new_channel_subset),
 520                "".join(new_channel_subset),
 521                len(sets_to_try),
 522                accuracy,
 523                precision,
 524                recall,
 525            ]
 526
 527        step += 1
 528
 529        stop_criterion = check_stopping_criterion(
 530            time.time() - start_time,
 531            len(new_channel_subset),
 532            p_delta,
 533            max_time,
 534            min_channels,
 535            max_channels,
 536            performance_delta,
 537            print_output=True,
 538        )
 539
 540    new_channel_subset = [channel_labels[c] for c in sfs_subset]
 541
 542    if print_output == "verbose" or print_output == "final":
 543        print(best_channel_subset)
 544        print(metric, " : ", best_performance)
 545        print("Time to optimal subset: ", time.time() - start_time, "s")
 546
 547    # Get the best model
 548
 549    return (
 550        best_channel_subset,
 551        best_model,
 552        best_preds,
 553        best_accuracy,
 554        best_precision,
 555        best_recall,
 556        results_df,
 557    )

The Sequential Forward Floating Selection (SFFS) method for channel selection.

Parameters
  • kernel_func (function): The classification kernel function which does feature extraction and classification. Different functions are used for MI, P300, SSVEP, etc.

  • X (numpy.ndarray): Training data for the classifier as windows of EEG data. 3D array containing data with float type.

    shape = (W_windows,C_channels,S_samples)

  • y (numpy.ndarray): Training labels for the classifier. 1D array.

    shape = (nwindows)

  • channel_labels (list of str): The set of channel labels corresponding to C_channels. A list of strings with length = C_channels.
  • metric (str): The metric used to measure the "goodness" of the trained classifier.
  • initial_channels (list of str): Initial guess of channels.
  • max_time (int): The maxiumum amount of time, in seconds, that the function will search for the optimal solution.
  • min_channels (int): The minimum number of channels.
  • max_channels (int): The maximum number of channels.
  • performance_delta (float): The performance delta under which the algorithm is considered to be close enough to optimal.
  • n_jobs (int): The number of threads to dedicate to this calculation.
  • print_output (str): Flag on whether or not to print output. Options are:
    • "verbose": Print output at each step.
    • "final": Print output at the end.
    • "silent": No output.
  • record_performance (bool): Flag on whether or not to record performance metrics at each step.
Returns
  • new_channel_subset (list of str): The new best channel subset from the list of channel_labels.
  • self.clf (classifier): The trained classification model.

  • preds (numpy.ndarray): The predictions from the model. 1D array with the same shape as y.

    shape = (nwindows)

  • accuracy (float): The accuracy of the trained classification model.
  • precision (float): The precision of the trained classification model.
  • recall (float): The recall of the trained classification model.
  • results_df (pandas.DataFrame): A dataframe containing the performance metrics at each step.
