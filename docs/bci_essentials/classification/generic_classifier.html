118    def get_subset(self, X=[]):
119        """Get a subset of X according to labels or indices.
120
121        Parameters
122        ----------
123        X : numpy.ndarray, *optional*
124            3D array containing data with `float` type.
125
126            shape = (`N_windows`,`M_channels`,`P_samples`)
127            - Default is `[]`.
128        subset : list of `int` or `str`, *optional*
129            List of indices (int) or labels (str) of the desired channels.
130            - Default is `[]`.
131        channel_labels : list of `str`, *optional*
132            Channel labels from the entire EEG montage.
133            - Default is `[]`.
134
135        Returns
136        -------
137        X : numpy.ndarray
138            Subset of input `X` according to labels or indices.
139            3D array containing data with `float` type.
140
141            shape = (`N_windows`,`M_channels`,`P_samples`)
142
143        """
144
145        # Check for self.subset and/or self.channel_labels
146
147        # Init
148        subset_indices = []
149
150        # Copy the indices based on subset
151        try:
152            # Check if we can use subset indices
153            if self.subset == []:
154                return X
155
156            if type(self.subset[0]) is int:
157                print("Using subset indices")
158
159                subset_indices = self.subset
160
161            # Or channel labels
162            if type(self.subset[0]) is str:
163                print("Using channel labels and subset labels")
164
165                # Replace indices with those described by labels
166                for sl in self.subset:
167                    subset_indices.append(self.channel_labels.index(sl))
168
169            # Return for the given indices
170            try:
171                # nwindows, nchannels, nsamples = self.X.shape
172
173                if X == []:
174                    new_X = self.X[:, subset_indices, :]
175                    self.X = new_X
176                else:
177                    new_X = X[:, subset_indices, :]
178                    X = new_X
179                    return X
180
181            except Exception:
182                # nchannels, nsamples = self.X.shape
183                if X == []:
184                    new_X = self.X[subset_indices, :]
185                    self.X = new_X
186
187                else:
188                    new_X = X[subset_indices, :]
189                    X = new_X
190                    return X
191
192        # notify if failed
193        except Exception:
194            print("something went wrong, no subset taken")
195            return X

313    def predict_decision_block(self, decision_block, print_predict=True):
314        """Predict a label based on a decision block.
315
316        Parameters
317        ----------
318        decision block : type
319            Description of parameter `decision block`.
320            If array, state size and type. E.g.
321            3D array containing data with `float` type.
322
323            shape = (`first_dimension`,`second_dimension`,`third_dimension`)
324        print_predict : type, *optional*
325            Description of parameter `print_predict`.
326            - Default is `True`.
327
328        Returns
329        -------
330        prediction : type
331            Description of returned object.
332
333        """
334        decision_block = self.get_subset(decision_block)
335
336        if print_predict:
337            print("making a prediction")
338
339        # get prediction probabilities for all
340        proba_mat = self.clf.predict_proba(decision_block)
341
342        proba = proba_mat[:, 1]
343
344        relative_proba = proba / np.amax(proba)
345
346        log_proba = np.log(relative_proba)
347
348        if print_predict:
349            print("log relative probabilities")
350            print(log_proba)
351
352        # the selection is the highest probability
353
354        prediction = int(np.where(proba == np.amax(proba))[0][0])
355
356        self.predictions.append(prediction)
357        self.pred_probas.append(proba_mat)
358
359        return prediction
