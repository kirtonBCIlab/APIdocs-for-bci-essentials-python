bci_essentials.eeg_data

Module for managing BCI data.

This module provides data classes for different BCI paradigms.

It includes the loading of offline data in xdf format or the live streaming of LSL data.

The loaded/streamed data is added to a buffer such that offline and online processing pipelines are identical.

Data is pre-processed (using the signal_processing module), windowed, and classified (using one of the classification sub-modules).

 175    def _pull_data_from_source(self):
 176        """Get pull data from EEG and optionally, the marker source.
 177
 178        This method will fill up the marker_data, eeg_data and corresponding timestamp arrays.
 179        """
 180        # pull from marker source if present
 181        if self.__marker_source is not None:
 182            new_marker_data, new_marker_timestamps = self.__marker_source.get_markers()
 183            self.marker_time_correction = self.__marker_source.time_correction()
 184
 185            # apply time correction
 186            new_marker_timestamps = [
 187                new_marker_timestamps[i] + self.marker_time_correction
 188                for i in range(len(new_marker_timestamps))
 189            ]
 190
 191            # save the marker data to the data object
 192            self.marker_data = np.array(list(self.marker_data) + new_marker_data)
 193            self.marker_timestamps = np.array(
 194                list(self.marker_timestamps) + new_marker_timestamps
 195            )
 196
 197        # pull from EEG source
 198        new_eeg_data, new_eeg_timestamps = self.__eeg_source.get_samples()
 199        new_eeg_data = np.array(new_eeg_data)
 200
 201        # Handle the case when you are using subsets
 202        if self.__subset != []:
 203            new_eeg_data = new_eeg_data[:, self.subset_indices]
 204
 205        # if time is in milliseconds, divide by 1000, works for sampling rates above 10Hz
 206        try:
 207            if self.time_units == "milliseconds":
 208                new_eeg_timestamps = [
 209                    (new_eeg_timestamps[i] / 1000)
 210                    for i in range(len(new_eeg_timestamps))
 211                ]
 212
 213        # If time units are not defined then define them
 214        except Exception:
 215            dif_low = -2
 216            dif_high = -1
 217            while new_eeg_timestamps[dif_high] - new_eeg_timestamps[dif_low] == 0:
 218                dif_low -= 1
 219                dif_high -= 1
 220
 221            if new_eeg_timestamps[dif_high] - new_eeg_timestamps[dif_low] > 0.1:
 222                new_eeg_timestamps = [
 223                    (new_eeg_timestamps[i] / 1000)
 224                    for i in range(len(new_eeg_timestamps))
 225                ]
 226                self.time_units = "milliseconds"
 227            else:
 228                self.time_units = "seconds"
 229
 230        # apply time correction, this is essential for headsets like neurosity which have their own clock
 231        self.eeg_time_correction = self.__eeg_source.time_correction()
 232
 233        # MAYBE DONT NEED THIS WITH NEW PROC SETTINGS
 234        new_eeg_timestamps = [
 235            new_eeg_timestamps[i] + self.eeg_time_correction
 236            for i in range(len(new_eeg_timestamps))
 237        ]
 238
 239        # save the EEG data to the data object
 240        try:
 241            self.eeg_data = np.concatenate((self.eeg_data, new_eeg_data))
 242        except Exception:
 243            self.eeg_data = new_eeg_data
 244
 245        # save the marker data to the data object
 246        self.eeg_timestamps = np.array(list(self.eeg_timestamps) + new_eeg_timestamps)
 247
 248        # If the outlet exists send a ping
 249        if self._messenger is not None:
 250            self.ping_count += 1
 251            if self.ping_count % self.ping_interval:
 252                self._messenger.ping()

 491    def __package_resting_state_data(self):
 492        """Package resting state data.
 493
 494        Returns
 495        -------
 496        `None`
 497            `self.rest_windows` is updated.
 498
 499        """
 500        try:
 501            logger.debug("Packaging resting state data")
 502
 503            eyes_open_start_time = []
 504            eyes_open_end_time = []
 505            eyes_closed_start_time = []
 506            eyes_closed_end_time = []
 507            rest_start_time = []
 508            rest_end_time = []
 509
 510            # Initialize start and end locations
 511            eyes_open_start_loc = []
 512            eyes_open_end_loc = []
 513            eyes_closed_start_loc = []
 514            eyes_closed_end_loc = []
 515            rest_start_loc = []
 516            rest_end_loc = []
 517
 518            current_time = self.eeg_timestamps[0]
 519            current_timestamp_loc = 0
 520
 521            for i in range(len(self.marker_data)):
 522                # Increment the EEG until just past the marker timestamp
 523                while current_time < self.marker_timestamps[i]:
 524                    current_timestamp_loc += 1
 525                    current_time = self.eeg_timestamps[current_timestamp_loc]
 526
 527                # get eyes open start times
 528                if self.marker_data[i][0] == "Start Eyes Open RS: 1":
 529                    eyes_open_start_time.append(self.marker_timestamps[i])
 530                    eyes_open_start_loc.append(current_timestamp_loc - 1)
 531                    logger.debug("received eyes open start")
 532
 533                # get eyes open end times
 534                if self.marker_data[i][0] == "End Eyes Open RS: 1":
 535                    eyes_open_end_time.append(self.marker_timestamps[i])
 536                    eyes_open_end_loc.append(current_timestamp_loc)
 537                    logger.debug("received eyes open end")
 538
 539                # get eyes closed start times
 540                if self.marker_data[i][0] == "Start Eyes Closed RS: 2":
 541                    eyes_closed_start_time.append(self.marker_timestamps[i])
 542                    eyes_closed_start_loc.append(current_timestamp_loc - 1)
 543                    logger.debug("received eyes closed start")
 544
 545                # get eyes closed end times
 546                if self.marker_data[i][0] == "End Eyes Closed RS: 2":
 547                    eyes_closed_end_time.append(self.marker_timestamps[i])
 548                    eyes_closed_end_loc.append(current_timestamp_loc)
 549                    logger.debug("received eyes closed end")
 550
 551                # get rest start times
 552                if self.marker_data[i][0] == "Start Rest for RS: 0":
 553                    rest_start_time.append(self.marker_timestamps[i])
 554                    rest_start_loc.append(current_timestamp_loc - 1)
 555                    logger.debug("received rest start")
 556                # get rest end times
 557                if self.marker_data[i][0] == "End Rest for RS: 0":
 558                    rest_end_time.append(self.marker_timestamps[i])
 559                    rest_end_loc.append(current_timestamp_loc)
 560                    logger.debug("received rest end")
 561
 562            # Eyes open
 563            # Get duration, nsmaples
 564
 565            if len(eyes_open_end_loc) > 0:
 566                duration = np.floor(eyes_open_end_time[0] - eyes_open_start_time[0])
 567                nsamples = int(duration * self.fsample)
 568
 569                self.eyes_open_timestamps = np.array(range(nsamples)) / self.fsample
 570                self.eyes_open_windows = np.ndarray(
 571                    (len(eyes_open_start_time), self.nchannels, nsamples)
 572                )
 573                # Now copy EEG for these windows
 574                for i in range(len(eyes_open_start_time)):
 575                    # For each channel of the EEG, interpolate to uniform sampling rate
 576                    for c in range(self.nchannels):
 577                        # First, adjust the EEG timestamps to start from zero
 578                        eeg_timestamps_adjusted = (
 579                            self.eeg_timestamps[
 580                                eyes_open_start_loc[i] : eyes_open_end_loc[i]
 581                            ]
 582                            - self.eeg_timestamps[eyes_open_start_loc[i]]
 583                        )
 584
 585                        # Second, interpolate to timestamps at a uniform sampling rate
 586                        channel_data = np.interp(
 587                            self.eyes_open_timestamps,
 588                            eeg_timestamps_adjusted,
 589                            self.eeg_data[
 590                                eyes_open_start_loc[i] : eyes_open_end_loc[i], c
 591                            ],
 592                        )
 593
 594                        # Third, add to the EEG window
 595                        self.eyes_open_windows[i, c, :] = channel_data
 596                        self.eyes_open_timestamps
 597
 598            logger.debug("Done packaging resting state data")
 599
 600            # Eyes closed
 601
 602            if len(eyes_closed_end_loc) > 0:
 603                # Get duration, nsmaples
 604                duration = np.floor(eyes_closed_end_time[0] - eyes_closed_start_time[0])
 605                nsamples = int(duration * self.fsample)
 606
 607                self.eyes_closed_timestamps = np.array(range(nsamples)) / self.fsample
 608                self.eyes_closed_windows = np.ndarray(
 609                    (len(eyes_closed_start_time), self.nchannels, nsamples)
 610                )
 611                # Now copy EEG for these windows
 612                for i in range(len(eyes_closed_start_time)):
 613                    # For each channel of the EEG, interpolate to uniform sampling rate
 614                    for c in range(self.nchannels):
 615                        # First, adjust the EEG timestamps to start from zero
 616                        eeg_timestamps_adjusted = (
 617                            self.eeg_timestamps[
 618                                eyes_closed_start_loc[i] : eyes_closed_end_loc[i]
 619                            ]
 620                            - self.eeg_timestamps[eyes_closed_start_loc[i]]
 621                        )
 622
 623                        # Second, interpolate to timestamps at a uniform sampling rate
 624                        channel_data = np.interp(
 625                            self.eyes_closed_timestamps,
 626                            eeg_timestamps_adjusted,
 627                            self.eeg_data[
 628                                eyes_closed_start_loc[i] : eyes_closed_end_loc[i], c
 629                            ],
 630                        )
 631
 632                        # Third, add to the EEG window
 633                        self.eyes_closed_windows[i, c, :] = channel_data
 634                        self.eyes_closed_timestamps
 635
 636            # Rest
 637            if len(rest_end_loc) > 0:
 638                # Get duration, nsmaples
 639                while rest_end_time[0] < rest_start_time[0]:
 640                    rest_end_time.pop(0)
 641                    rest_end_loc.pop(0)
 642
 643                duration = np.floor(rest_end_time[0] - rest_start_time[0])
 644
 645                nsamples = int(duration * self.fsample)
 646
 647                self.rest_timestamps = np.array(range(nsamples)) / self.fsample
 648                self.rest_windows = np.ndarray(
 649                    (len(rest_start_time), self.nchannels, nsamples)
 650                )

 194            self.marker_timestamps = np.array(
 195                list(self.marker_timestamps) + new_marker_timestamps
 196            )
 197
 198        # pull from EEG source
 199        new_eeg_data, new_eeg_timestamps = self.__eeg_source.get_samples()
 200        new_eeg_data = np.array(new_eeg_data)
 201
 202        # Handle the case when you are using subsets
 203        if self.__subset != []:
 204            new_eeg_data = new_eeg_data[:, self.subset_indices]
 205
 206        # if time is in milliseconds, divide by 1000, works for sampling rates above 10Hz
 207        try:
 208            if self.time_units == "milliseconds":
 209                new_eeg_timestamps = [
 210                    (new_eeg_timestamps[i] / 1000)
 211                    for i in range(len(new_eeg_timestamps))
 212                ]
 213
 214        # If time units are not defined then define them
 215        except Exception:
 216            dif_low = -2
 217            dif_high = -1
 218            while new_eeg_timestamps[dif_high] - new_eeg_timestamps[dif_low] == 0:
 219                dif_low -= 1
 220                dif_high -= 1
 221
 222            if new_eeg_timestamps[dif_high] - new_eeg_timestamps[dif_low] > 0.1:
 223                new_eeg_timestamps = [
 224                    (new_eeg_timestamps[i] / 1000)
 225                    for i in range(len(new_eeg_timestamps))
 226                ]
 227                self.time_units = "milliseconds"
 228            else:
 229                self.time_units = "seconds"
 230
 231        # apply time correction, this is essential for headsets like neurosity which have their own clock
 232        self.eeg_time_correction = self.__eeg_source.time_correction()
 233
 234        # MAYBE DONT NEED THIS WITH NEW PROC SETTINGS
 235        new_eeg_timestamps = [
 236            new_eeg_timestamps[i] + self.eeg_time_correction
 237            for i in range(len(new_eeg_timestamps))
 238        ]
 239
 240        # save the EEG data to the data object
 241        try:
 242            self.eeg_data = np.concatenate((self.eeg_data, new_eeg_data))
 243        except Exception:
 244            self.eeg_data = new_eeg_data
 245
 246        # save the marker data to the data object
 247        self.eeg_timestamps = np.array(list(self.eeg_timestamps) + new_eeg_timestamps)
 248
 249        # If the outlet exists send a ping
 250        if self._messenger is not None:
 251            self.ping_count += 1
 252            if self.ping_count % self.ping_interval:
 253                self._messenger.ping()

 493        """Package resting state data.
 494
 495        Returns
 496        -------
 497        `None`
 498            `self.rest_windows` is updated.
 499
 500        """
 501        try:
 502            logger.debug("Packaging resting state data")
 503
 504            eyes_open_start_time = []
 505            eyes_open_end_time = []
 506            eyes_closed_start_time = []
 507            eyes_closed_end_time = []
 508            rest_start_time = []

 520            current_timestamp_loc = 0
 521

 523                # Increment the EEG until just past the marker timestamp
 524                while current_time < self.marker_timestamps[i]:
 525                    current_timestamp_loc += 1

 527

 530                    eyes_open_start_time.append(self.marker_timestamps[i])

 572                    (len(eyes_open_start_time), self.nchannels, nsamples)

 576                    # For each channel of the EEG, interpolate to uniform sampling rate

 581                                eyes_open_start_loc[i] : eyes_open_end_loc[i]

 583                            - self.eeg_timestamps[eyes_open_start_loc[i]]
 584                        )

 588                            self.eyes_open_timestamps,
 589                            eeg_timestamps_adjusted,

 591                                eyes_open_start_loc[i] : eyes_open_end_loc[i], c
 592                            ],
 593                        )

 595                        # Third, add to the EEG window

 656                        # First, adjust the EEG timestamps to start from zero

 658                            self.eeg_timestamps[rest_start_loc[i] : rest_end_loc[i]]
 659                            - self.eeg_timestamps[rest_start_loc[i]]

Main function of EEG_data class.

Runs a while loop that reads in EEG data from the EEG_data object and processes it. Can be used in online or offline mode.

  • If in online mode, then the loop will continuously try to read in data from the EEG_data object and process it. The loop will terminate when max_loops is reached, or when manually terminated.
  • If in offline mode, then the loop will read in all of the data at once, process it, and then terminate.
